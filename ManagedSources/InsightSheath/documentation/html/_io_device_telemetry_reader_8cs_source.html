510 public static IoDeviceTelemetryNtCreateFile GetNtCreateFileSettings(this DebugEventExceptionInfo that)
511 {
512 IoDeviceTelemetryNtCreateFile ret;
513 MachineType Type;
514 IntPtr Handle = HelperRoutines.OpenProcessForVirtualMemory(that.ProcessID);
515 var arguments = that.ExceptionParameter64;
516 if (that.IsEventFrom32BitProcess)
517 {
518 Type = MachineType.MachineI386;
519 }
520 else
521 {
522 Type = MachineType.MachineAmd64;
523 }
524 try
525 {
526 ret = new IoDeviceTelemetryNtCreateFile(that.ProcessID, that.ThreadID, new IntPtr((long)arguments[NtCreateFile_ReturnHandle]), new IntPtr((long)arguments[GeneralTelemetry.LastError_Ptr]), Type) ;
527// ret.ReturnValue = new IntPtr((long)arguments[LastError_Ptr]);
528 // ret.FileOutHandle = new IntPtr((long)arguments[NtCreateFile_ReturnHandle]);
529 ret.DesiredAccess = (AccessMasks) arguments[NtCreateFile_DesiredAccess];
530 if (arguments[NtCreateFile_ObjectAttributes] == 0)
531 {
532 ret.ObjectAttributes = null;
533 }
534 else
535 {
536 //ret.ObjectAttributes = new Structs.WindowsObjectAttributes(new IntPtr((long)arguments[NtCreateFile_ObjectAttributes]));
537 if (arguments[NtCreateFile_ObjectAttributes] != 0)
538 {
539 ret.ObjectAttributes = RemoteStructure.RemoteReadObjectAttributes(Handle, new IntPtr((long)arguments[NtCreateFile_ObjectAttributes]), that.IsEventFrom32BitProcess, true);
540 }
541 else
542 {
543 ret.ObjectAttributes = null;
544 }
545
546 }
547 if (arguments[NtCreateFile_AllocationSize] != 0)
548 {
549 ret.AllocationSize = Marshal.PtrToStructure<LARGE_INTEGER>(new IntPtr((long)arguments[NtCreateFile_AllocationSize]));
550 }
551 else
552 {
553 ret.AllocationSize = new LARGE_INTEGER();
554 }
555 ret.FileAttributes = (FileAttributes) arguments[NtCreateFile_FileAttributs];
556 ret.ShareAccess = (FileShare) arguments[NtCreateFile_ShareAccess];
557
558 ret.CreateDisposition = (NtCreationDisposition) arguments[NtCreateFile_CreateDisposition];
559 ret.CreationOptions = (uint)arguments[NtCreateFile_CreateOptions];
560 ret.EaBuffer = new IntPtr((long)arguments[NtCreateFile_EaBuffer]);
561 ret.EaSize = arguments[NtCreateFile_EaLength];
562 //ret.ForceHandle = new IntPtr((long)arguments[NtCreateFile_OverwriteHandle]);
563 return ret;
564 }
565 finally
566 {
567 HelperRoutines.CloseHandle(Handle);
568 }
